04. Higher-Order Functions

ND032 C02 L03 A03 Flexible Programs With Higher Order Functions

Code Review

Code Review

function greaterThan(n) {
    return m => m > n;
}

let greaterThanEight = greaterThan(8);
console.log(greaterThanEight)
// expected output:  [Function]

console.log(greaterThanEight(11));
// expected output: true

// The syntax above can be tricky, but it is all ES6

// Equivalent to:
function greaterThan(n) {
    return function (m) {
        return m > n;
    }
}
